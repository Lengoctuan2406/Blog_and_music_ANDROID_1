Table users {
  	user_id varchar pk
  
  	account_name varchar
  	email varchar
  	password varchar
  	date_of_birth datetime
  	gender varchar
  	phone varchar
  	avatar varchar
  	facebook varchar
  	instagram varchar

  	role varchar
}

Table hotels {
  	hotel_id varchar pk

  	name varchar
  	description varchar
	country_id varchar
	address varchar
  	picture_1 varchar
  	picture_2 varchar
  	picture_3 varchar
	hotel_details_id varchar
	number_days_of_stay_id varchar
	star int

	user_id varchar
	status varchar//0 là chưa like, 1 là đã like
}

Table number_days_of_stay { //số ngày ở lại
  	number_days_of_stay_id varchar pk

	description varchar //Mô tả, ví dụ 1N-1D
	description_number int //Mô tả dạng số
}

Table hotel_details { //trang Choose your stay
  	hotel_details_id varchar pk
  
  	number_of_room_id varchar

	size varchar //độ rộng của phòng
	description varchar //tự nhập(cho bao nhiêu người, còn gì khác)
	price int (dạng dollars) (tự cho)
	status varchar
}

Table number_of_room {
  	number_of_room_id varchar pk

  	description varchar //có thể Double room, Single room, Tripper room
}
